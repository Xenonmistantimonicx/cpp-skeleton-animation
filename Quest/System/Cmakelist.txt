cmake_minimum_required(VERSION 3.16)
project(SoloLevelingQuestSystem CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

if(MSVC)
    add_compile_options(/W4 /utf-8)
else()
    add_compile_options(-Wall -Wextra -O2)
endif()

add_executable(solo_leveling
    main.cpp
    QuestSystem.cpp
)

target_include_directories(solo_leveling PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})
